스택은 LIFO(Last In First Out)의 구조로 나중에 들어온 것이 먼저 나가는 자료구조이다.
여러 방법으로 비유하지만 글쓴이는 프링글스 감자칩에 비유한다. 우리가 제일 처음 통에 들어온 감자칩을 먹기 위해선 위에 것들 부터 꺼내야 한다.

```{.cpp}
#include <stdio.h>
#define MAX 100

int stack[MAX];
int topIdx = -1;

int IsEmpty() {
	if (topIdx == -1) {
		return true;
	}
	else {
		return false;
	}
}

void Push(int data) {
	if (topIdx < MAX) {
		stack[++topIdx] = data;
		printf("%d Push!\n", stack[topIdx]);
	}
}

void Pop() {
	if (IsEmpty) {
		printf("%d Pop!\n", stack[topIdx]);
		stack[topIdx] = NULL;
		topIdx--;
	}
}

int main() {
	Push(1);
	Push(2);
	Push(3);
	Push(7);
	Push(9);
	Pop();
	Pop();
	int i = 0;
	printf("현재 스택 상황: ");
	while (stack[i] != NULL) {
		printf("%d ", stack[i]);
		i++;
	}
}
```
