int main() {
  int price = 770;
  int change = 0;
  int coin[] = { 500,100,50,10 };
  
  for (int i = 0; i < sizeof(coin) / sizeof(int); i++) {
    change += price / coin[i];
    price = price % coin[i];
  }
  printf("거스름돈 동전 갯수는: %d", change);
}
